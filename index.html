<script>
        var num1, num2;
        var result;

        function getNum1() {
            num1 = document.getElementById('num1').value;
            return Number(num1);
        }
        function getNum2() {
            num2 = document.getElementById('num2').value;
            return Number(num2);
        }
        function setResult() {
            document.getElementById('result').value = result;
        }

        function doCalc() {
            num1 = getNum1();
            num2 = getNum2();

            var plus = document.getElementById('plus').checked;
            var minus = document.getElementById('minus').checked;
            var divide = document.getElementById('divide').checked;
            var multiply = document.getElementById('multiply').checked;

            if(multiply){
                result = num1 * num2;
            }
            else if(divide && (num1 != 0) && (num2 == 0)){
                result = "No division by 0!";
            }else if(divide) {
                result = num1 / num2;
            }else if (minus) {
                result = num1 - num2;
            }else {
                result = num1 + num2;
            }

            setResult();
        }

    var btn = document.getElementById("btn");
    btn.addEventListener("click", doCalc);

</script>
